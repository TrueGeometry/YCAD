# Tweezers Design Script
/view top

# 1. Hinge
/parametric cylinder 0.5 0.5 1
/rotate 90 0 0
/move -5 0 0
/tag_last Hinge

# 2. Arms
/parametric box 10 0.2 0.8
/rotate 0 0 5
/move 0 0.5 0
/tag_last TopArm
/parametric box 10 0.2 0.8
/rotate 0 0 -5
/move 0 -0.5 0
/tag_last BotArm

# 3. Union
/union @Hinge @TopArm @BotArm
/tag_last TweezerBody

# 4. Finish
/setprop @TweezerBody material stainless_steel
/view iso
/view fit
